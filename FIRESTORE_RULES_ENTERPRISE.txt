rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    function isSignedIn() {
      return request.auth != null;
    }
    function isRole(role) {
      return isSignedIn() && request.auth.token.role == role;
    }
    function isGoogleProvider() {
      return isSignedIn() && request.auth.token.firebase.sign_in_provider == 'google.com';
    }
    function isStaffEmail() {
      return isGoogleProvider() && request.auth.token.email in [
        'marco.tranquilli@dos.design',
        'lorenzo.russo@alimentarirusso',
        'russolorenzo11@gmail.com',
        'beatrice.binini@dos.design',
        'monica.porta@dos.design'
      ];
    }
    function isNamedRistoratore() {
      return isGoogleProvider()
        && request.auth.token.email_verified == true
        && request.auth.token.name in ['Lorenzo Russo', 'lorenzo russo'];
    }
    function isAdminEmail() {
      return isGoogleProvider() && request.auth.token.email == 'marco.tranquilli@dos.design';
    }
    function isStaff() {
      return isSignedIn() && (request.auth.token.role in ['admin','ristoratore','facility']);
    }
    function isStaffIdentity() {
      return isStaff() || isStaffEmail() || isNamedRistoratore();
    }
    function isOwnerByUid() {
      return isSignedIn() && resource.data.uid == request.auth.uid;
    }
    function isCreateOwnerByUid() {
      return isSignedIn() && request.resource.data.uid == request.auth.uid;
    }

    // Ordini:
    // - utenti vedono i propri ordini (uid)
    // - staff vede tutti gli ordini
    // - create consentita solo per il proprio uid
    match /orders/{doc} {
      allow read: if isStaffIdentity() || isOwnerByUid();
      allow create: if isCreateOwnerByUid();
      allow update: if isStaffIdentity();
      allow delete: if isRole('admin') || isAdminEmail();
    }

    // Frige prodotti: gestione staff
    match /frige_products/{doc} {
      allow read: if isStaffIdentity();
      allow write: if isStaffIdentity();
    }
    // Frige acquisti:
    // - utente crea e legge i propri acquisti
    // - staff legge/aggiorna tutti
    match /frige_purchases/{doc} {
      allow read: if isStaffIdentity() || isOwnerByUid();
      allow create: if isCreateOwnerByUid();
      allow update: if isStaffIdentity();
    }
    match /frige_refills/{doc} {
      allow read, write: if isStaffIdentity();
    }

    // Disponibilità menu: solo admin/ristoratore scrivono
    match /menu_products/{doc} {
      allow read: if isSignedIn();
      allow write: if isRole('admin') || isRole('ristoratore') || isStaffEmail() || isNamedRistoratore();
    }

    // Storico creazioni custom e voti
    match /custom_creations/{doc} {
      allow read: if isSignedIn();
      allow create: if isCreateOwnerByUid();
      allow update: if isOwnerByUid() || (
        isSignedIn() &&
        request.resource.data.uid == resource.data.uid &&
        request.resource.data.email == resource.data.email &&
        request.resource.data.createdAt == resource.data.createdAt &&
        request.resource.data.diff(resource.data).changedKeys().hasOnly(['votes','voters','updatedAt'])
      );
      allow delete: if isRole('admin');
    }

    // Audit menu: solo staff
    match /menu_audit/{doc} {
      allow read, write: if isStaffIdentity();
    }

    // Audit ordini: solo staff
    match /orders_audit/{doc} {
      allow read, write: if isStaffIdentity();
    }
  }
}
